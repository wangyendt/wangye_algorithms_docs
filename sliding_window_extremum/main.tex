\documentclass{article}
\usepackage[ruled,linesnumbered]{algorithm2e}
\SetKwFunction{PeekLeft}{peekleft()}
\SetKwFunction{PeekRight}{peekright()}
\SetKwFunction{PopLeft}{popleft()}
\SetKwFunction{PopRight}{popright()}
\SetKwFunction{Append}{append}
\SetKw{Yield}{yield}
\begin{document}
    \begin{algorithm}[H]
        \caption{sliding window extremum}
        \KwData{sequence $X_1,...,X_N$, sliding window length $L$}
        \KwResult{local minimum $m_1$,...,$m_N$}
        Initialize local minimum double-ended queue $Q$\;
        \For{$i\leftarrow 1$ \KwTo $N$}{
            \uIf{$Q\neq\emptyset$ and $Q.\PeekLeft\geq L$}{
                $Q.\PopLeft$
            }
            \While{$Q\neq\emptyset$ and $X_{Q.\PeekRight}$}{
                $Q.\PopRight$
            }
            $Q.\Append{i}$\;
            $\Yield$ $m_i=X_{Q.\PeekLeft}$\;
        }
    \end{algorithm}
\end{document}
